$def with (totalPosts, totalPostsPublished, unpublishedPosts,totalPages, totalPagesPublished, unpublishedPages,totalTopics, totalTopicsPublished,totaltopicsUnpublished,unpublishedTopics) $var title ="Dashboard"

Blog Posts

  • $:totalPosts Total Posts
  • $:totalPostsPublishedPublished Posts
  • $:(totalPosts - totalPostsPublished) Un Published Posts
$if unpublishedPosts:

Un Published Posts

$for post in unpublishedPosts:
TITLE CREATED ON PUBLISHED ON EDIT DELETE
$:post.title $:post.created_at.strftime("%B %d, %Y") $:post.published_at.strftime("%B %d, %Y") EDIT DELETE

Blog Pages

  • $:totalPages Total Pages
  • $:totalPagesPublishedPublished Pages
  • $:(totalPages - totalPagesPublished) Un Published Pages
$if unpublishedPages:

Un Published Pages

$for page in unpublishedPages:
TITLE CREATED ON PUBLISHED ON EDIT DELETE
$:page.title $:page.created_at.strftime("%B %d, %Y") $:page.published_at.strftime("%B %d, %Y") EDIT DELETE

Blog Topics

  • $:totalTopics Total Topics
  • $:totalTopicsPublishedPublished Topics
  • $:totaltopicsUnpublished Un Published Topics
  • $:(totalTopics -(totalTopicsPublished - totaltopicsUnpublished ))UN REFERENCED Topics
$if unpublishedTopics:

Un Published Topics

$for topic in unpublishedTopics
TITLE SLUG POSTS COUNT EDIT DELETE
$:topic.title $:topic.slug $:topic.posts_count EDIT DELETE
